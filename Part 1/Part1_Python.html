#parameters
x0 = 0 #initial value of x
N = 10000
s = 1

np.random.seed(123) #lock the random generation - replicable just like the report 

#pdf given
def pdf(x):
    return 0.5 * np.exp(-np.abs(x))

#random walk metropolis given
def metropolis(x0, N, s):
    x = np.zeros(N) #create array x to store the chain
    x[0] = x0 #first value is x0

    for i in range(1, N):
        random_number = np.random.normal(loc=x[i-1], scale=s) #new number is selected from ND centered at the previous x value
        r = pdf(random_number) / pdf(x[i-1]) #calculate r
        
        if np.log(np.random.uniform()) < np.log(r): #generate random u from UD, if logu < log r
            x[i] = random_number #new xi is the random number
        else:
            x[i] = x[i-1] #else same as last
    
    return x

#get samples and kernel density
samples = metropolis(x0, N, s)
kernel_density = gaussian_kde(samples)

#plots
x_vals = np.linspace(-5, 5, 1000)

plt.figure(figsize=(10, 6))
plt.hist(samples, bins=50, density=True, color="#818aa3", edgecolor='white', alpha=0.8, label="Samples Density Distribution")
plt.plot(x_vals, kernel_density(x_vals), color="#6d2e46", linewidth=2, label="Kernel Density Line")
plt.plot(x_vals, pdf(x_vals), color="#6adc8e", linewidth=2, label="Probability Density Function")
plt.xlim(-5, 5)
plt.ylim(0.0, 0.6)
plt.xticks(np.arange(-5, 6, 1))
plt.title("Random Walk Metropolis Samples")
plt.xlabel("x")
plt.legend(loc="upper right")
plt.grid(alpha=0.2)
plt.tight_layout()
plt.show()

# Monte Carlo estimates
mean_estimate = np.mean(samples)
sd_estimate = np.std(samples)

print("Monte Carlo Mean Estimate:", mean_estimate)
print("Monte Carlo Standard Deviation Estimate:", sd_estimate)

#function to calculate Rhat

def calculate_r_hat(chains):
    J = len(chains) #number of chains
    N = len(chains[0]) #number of iterations

    means = np.array([np.mean(chain) for chain in chains]) #mean of each array
    variances = np.array([np.var(chain, ddof=1) for chain in chains]) #var of each array

    W = np.mean(variances)
    M = np.mean(means)
    B = np.mean((means - M)**2)

    R_hat = np.sqrt((B + W) / W)
    return R_hat

#values of N and s given
N = 2000
s = 0.001
x0_initial_values = [0, 1, 2, 3]  #considered 4 chains with different x0 initial values


chains = [metropolis(x0, N, s) for x0 in x0_initial_values] #run metropolis 4 times
R_hat = calculate_r_hat(chains)
print(f"R-hat for s = 0.001: {R_hat:.4f}")

#create plot for values of Rhat over grid of s values
s_values = np.arange(0.001, 1.001, 0.001) #create values from 0.001 to 1.000, increasing by 0.001
R_hats = []

for s_val in s_values: #repeat the exercise for each value of s, and add it to r_hats
    chains = [metropolis(x0, N, s_val) for x0 in x0_initial_values]
    R_hats.append(calculate_r_hat(chains))

#turn into df
r_hat_df = pd.DataFrame({
    's': s_values,
    'R_hat': R_hats
})

r_hat_df['threshold_group'] = np.where(r_hat_df['R_hat'] > 1.05, "Above 1.05", "Below or 1.05") #create new column
